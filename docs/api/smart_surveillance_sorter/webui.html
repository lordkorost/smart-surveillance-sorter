  28def get_system_info():
  29    """Collect and format system information for display (CPU, RAM, GPU, Ollama status).
  30    
  31    Returns:
  32        Formatted string with system specifications
  33    """
  34    import platform, psutil, torch
  35    cpu = platform.processor() or platform.machine()
  36    ram_used = psutil.virtual_memory().used / 1024**3
  37    ram_total = psutil.virtual_memory().total / 1024**3
  38    
  39    lines = [
  40        f"**CPU:** {cpu}",
  41        f"**RAM:** {ram_used:.1f} / {ram_total:.1f} GB",
  42        f"**Python:** {platform.python_version()}",
  43        f"**PyTorch:** {torch.__version__}",
  44    ]
  45    
  46    if torch.cuda.is_available():
  47        gpu = torch.cuda.get_device_name(0)
  48        vram_used = torch.cuda.memory_allocated(0) / 1024**3
  49        vram_total = torch.cuda.get_device_properties(0).total_memory / 1024**3
  50        lines.append(f"**GPU:** {gpu}")
  51        lines.append(f"**VRAM:** {vram_used:.1f} / {vram_total:.1f} GB")
  52    else:
  53        lines.append("**GPU:** Not detected (CPU mode)")
  54    
  55    # Ollama check
  56    try:
  57        vision_cfg = load_json(SETTINGS_JSON).get("vision_settings", {})
  58        if validate_ollama_setup(vision_cfg):
  59            model = vision_cfg.get("model_name", "unknown")
  60            lines.append(f"**Ollama:** ✅ Connected ({model})")
  61        else:
  62            lines.append("**Ollama:** ❌ Not available")
  63    except:
  64        lines.append("**Ollama:** ❌ Not available")
  65    
  66    return "\n\n".join(lines)

 344def save_single_camera(cam_id, name, loc, patterns, priority, desc, dynamic, ignore,
 345                       th_p, th_v, th_a, nth_p, nth_v, nth_a,
 346                       fw_ground, fw_garden, fw_shoe, fw_wood,
 347                       thr_person, thr_vehicle, thr_animal,
 348                       fpw_person, fpw_animal, fpw_vehicle):
 349    """Save individual camera configuration.
 350    
 351    Builds FAKE_WEIGHTS with only values >= 0 (omits keys set to -1).
 352    
 353    Args:
 354        cam_id: Camera ID
 355        name: Camera name
 356        loc: Camera location
 357        patterns: Comma-separated search patterns
 358        priority: Priority category
 359        desc: Camera description
 360        dynamic: Whether to use dynamic stride
 361        ignore: Comma-separated ignore labels
 362        th_p/th_v/th_a: Day thresholds for person/vehicle/animal
 363        nth_p/nth_v/nth_a: Night thresholds
 364        fw_ground/fw_garden/fw_shoe/fw_wood: Fake weight values
 365        thr_person/thr_vehicle/thr_animal: BLIP threshold overrides
 366        fpw_person/fpw_animal/fpw_vehicle: Fake penalty weight overrides
 367        
 368    Returns:
 369        Status message
 370    """
 371    cameras = load_json(CAMERAS_JSON)
 372
 373    thresholds_night = {}
 374    if float(nth_p) >= 0: thresholds_night["person"]  = float(nth_p)
 375    if float(nth_v) >= 0: thresholds_night["vehicle"] = float(nth_v)
 376    if float(nth_a) >= 0: thresholds_night["animal"]  = float(nth_a)
 377
 378    threshold_override = {}
 379    if float(thr_person)  >= 0: threshold_override["PERSON"]  = float(thr_person)
 380    if float(thr_vehicle) >= 0: threshold_override["VEHICLE"] = float(thr_vehicle)
 381    if float(thr_animal)  >= 0: threshold_override["ANIMAL"]  = float(thr_animal)
 382
 383    penalty_override = {}
 384    if float(fpw_person)  >= 0: penalty_override["PERSON"]  = float(fpw_person)
 385    if float(fpw_animal)  >= 0: penalty_override["ANIMAL"]  = float(fpw_animal)
 386    if float(fpw_vehicle) >= 0: penalty_override["VEHICLE"] = float(fpw_vehicle)
 387
 388    # FAKE_WEIGHTS: save only values >= 0, omit keys set to -1
 389    fake_weights = {}
 390    if float(fw_ground) >= 0: fake_weights["GROUND"] = float(fw_ground)
 391    if float(fw_garden) >= 0: fake_weights["GARDEN"] = float(fw_garden)
 392    if float(fw_shoe)   >= 0: fake_weights["SHOE"]   = float(fw_shoe)
 393    if float(fw_wood)   >= 0: fake_weights["WOOD"]   = float(fw_wood)
 394
 395    blip_rules = {"FAKE_WEIGHTS": fake_weights}
 396    if threshold_override: blip_rules["THRESHOLD"] = threshold_override
 397    if penalty_override:   blip_rules["FAKE_PENALTY_WEIGHT"] = penalty_override
 398
 399    cam_data = {
 400        "name": name, "location": loc,
 401        "search_patterns": [p.strip() for p in patterns.split(",") if p.strip()],
 402        "priority": priority, "desc": desc, "dynamic_stride": dynamic,
 403        "filters": {"ignore_labels": [i.strip() for i in ignore.split(",") if i.strip()]},
 404        "thresholds": {"person": float(th_p), "vehicle": float(th_v), "animal": float(th_a)},
 405        "blip_rules": blip_rules,
 406    }
 407    if thresholds_night:
 408        cam_data["thresholds_night"] = thresholds_night
 409
 410    cameras[cam_id] = cam_data
 411    save_json(cameras, CAMERAS_JSON)
 412    return f"✅ Cam {cam_id} ({name}) saved!"

 435def delete_camera(cam_id):
 436    """Delete a camera configuration.
 437    
 438    Args:
 439        cam_id: Camera ID to delete
 440        
 441    Returns:
 442        Tuple of (updated dropdown, status message)
 443    """
 444    if not cam_id:

 446    cameras = load_json(CAMERAS_JSON)

 448        return gr.update(), "⚠️ Cannot delete the last camera."

 452        save_json(cameras, CAMERAS_JSON)

 454        return gr.update(choices=new_ids, value=new_ids[0] if new_ids else None), f"🗑️ {cam_id} ({cam_name}) deleted."
 455    return gr.update(), "❌ Camera not found."

 549class QueueHandler(logging.Handler):
 550    """Logging handler that queues log records to a thread-safe queue.
 551    
 552    Used to stream logs from worker processes to the WebUI.
 553    """
 554    def __init__(self, log_queue):
 555        """Initialize handler with a log queue.
 556        
 557        Args:
 558            log_queue: Queue to put log records into
 559        """
 560        super().__init__()
 561        self.log_queue = log_queue
 562
 563    def emit(self, record):
 564        """Emit a log record by adding it to the queue.
 565        
 566        Args:
 567            record: LogRecord to emit
 568        """
 569        self.log_queue.put(record)
 570        try:
 571            self.log_queue.put_nowait(self.format(record))
 572        except Exception:
 573            pass

 575def _run_scanner(input_path, output_path, mode, model_name,
 576                 use_refine, engine, use_fallback, is_clean_check,
 577                 no_sort, test_mode, device):
 578    """Start Scanner in thread and return (thread, result_holder, log_queue, q_handler)."""
 579    # Update model and device in settings.json
 580    try:
 581        with open(SETTINGS_JSON, "r", encoding="utf-8") as f:
 582            settings_dict = json.load(f)
 583        settings_dict.setdefault("yolo_settings", {})
 584        settings_dict["yolo_settings"]["model_path"] = model_name.replace(".pt", "")
 585        settings_dict["yolo_settings"]["device"]     = device or "cuda"
 586        with open(SETTINGS_JSON, "w", encoding="utf-8") as f:
 587            json.dump(settings_dict, f, indent=4, ensure_ascii=False)
 588    except Exception as e:
 589        log.warning(f"Error updating settings.json: {e}")
 590
 591    log_queue   = queue.Queue()
 592    q_handler   = QueueHandler(log_queue)
 593    q_handler.setFormatter(logging.Formatter("%(asctime)s %(levelname)-7s : %(message)s", "%H:%M:%S"))
 594    logging.getLogger().addHandler(q_handler)
 595
 596    result_holder = {"error": None, "done": False, "scanner": None, "start_time": time.time()}
 597
 598    def _run():
 599        try:
 600            get_logger(debug=test_mode)
 601            scanner = Scanner(
 602                mode=mode, device=device or None,
 603                is_refine=use_refine, is_fallback=use_fallback,
 604                is_test=test_mode, engine=engine,
 605                is_check_clean=is_clean_check, is_sort=not no_sort,
 606            )
 607            result_holder["scanner"] = scanner
 608            scanner.scan_folder(input_path, output_path)
 609        except Exception as e:
 610            result_holder["error"] = f"{str(e)}\n{traceback.format_exc()}"
 611        finally:
 612            result_holder["done"] = True
 613
 614    thread = threading.Thread(target=_run, daemon=True)
 615    thread.start()
 616    return thread, result_holder, log_queue, q_handler

 619def _stream_logs(thread, result_holder, log_queue, q_handler, header=""):
 620    """Generator that streams logs to the UI."""
 621    log_lines = [header] if header else []
 622    while not result_holder["done"] or not log_queue.empty():
 623        try:
 624            line = log_queue.get(timeout=0.1)
 625            log_lines.append(line)
 626        except queue.Empty:
 627            pass
 628        yield "\n".join(log_lines[-80:])
 629        time.sleep(0.1)
 630
 631    logging.getLogger().removeHandler(q_handler)
 632    
 633    scanner = result_holder.get("scanner")
 634    summary = ""
 635    if scanner and hasattr(scanner, "_get_final_summary"):
 636        try:
 637            elapsed = time.time() - result_holder.get("start_time", time.time())
 638            summary = "\n" + "─" * 50 + "\n" + scanner._get_final_summary(elapsed) + "\n" + "─" * 50
 639        except Exception:
 640            pass
 641
 642    if result_holder["error"]:
 643        yield "\n".join(log_lines) + f"\n\n💥 ERROR:\n{result_holder['error']}"
 644    else:
 645        yield "\n".join(log_lines) + summary + "\n\n✅ COMPLETED!!"

 724def run_realtime(input_path, output_path, mode, model_name, engine, device, interval):
 725    global _rt_thread, _rt_stop_event
 726
 727    if _rt_thread and _rt_thread.is_alive():
 728        yield "⚠️ Real-time already running. Stop it before restarting."
 729        return
 730    if not input_path:
 731        yield "⚠️ Specify the input folder."
 732        return
 733    input_p = Path(input_path)
 734    if not input_p.exists() or not input_p.is_dir():
 735        yield f"❌ Path not valid: {input_path}"
 736        return
 737
 738    final_output = output_path.strip() if output_path and output_path.strip() else input_path
 739    _rt_stop_event.clear()
 740
 741    log_queue = queue.Queue()
 742    q_handler = QueueHandler(log_queue)
 743    q_handler.setFormatter(logging.Formatter("%(asctime)s %(levelname)-7s : %(message)s", "%H:%M:%S"))
 744    logging.getLogger().addHandler(q_handler)
 745
 746    def _loop():
 747        cycle = 0
 748        while not _rt_stop_event.is_set():
 749            cycle += 1
 750            log.info(f"─── Cycle #{cycle} ───")
 751            try:
 752                get_logger(debug=False)
 753                with open(SETTINGS_JSON, "r", encoding="utf-8") as f:
 754                    s = json.load(f)
 755                s.setdefault("yolo_settings", {})
 756                s["yolo_settings"]["model_path"] = model_name.replace(".pt", "")
 757                s["yolo_settings"]["device"]     = device or "cuda"
 758                with open(SETTINGS_JSON, "w", encoding="utf-8") as f:
 759                    json.dump(s, f, indent=4, ensure_ascii=False)
 760
 761                scanner = Scanner(
 762                    mode=mode, device=device or None,
 763                    is_refine=True, is_fallback=False,
 764                    is_test=False, engine=engine,
 765                    is_check_clean=False, is_sort=True,
 766                )
 767                scanner.scan_folder(input_path, final_output)
 768            except RuntimeError as e:
 769                log.critical(f"Error: {e}")
 770                break
 771            except Exception as e:
 772                log.error(f"Error cycle #{cycle}: {e}")
 773
 774            if not _rt_stop_event.is_set():
 775                log.info(f"Cycle #{cycle} completed. Next in {interval}s")
 776                _rt_stop_event.wait(timeout=interval)
 777
 778        log.info("Real-time sorter stopped.")
 779
 780    _rt_thread = threading.Thread(target=_loop, daemon=True)
 781    _rt_thread.start()
 782
 783    log_lines = []
 784    while _rt_thread.is_alive() or not log_queue.empty():
 785        try:
 786            line = log_queue.get(timeout=0.1)
 787            log_lines.append(line)
 788        except queue.Empty:
 789            pass
 790        yield "\n".join(log_lines[-80:])
 791        time.sleep(0.1)
 792
 793    logging.getLogger().removeHandler(q_handler)
 794    yield "\n".join(log_lines) + "\n\n🛑 Real-time stopped."
