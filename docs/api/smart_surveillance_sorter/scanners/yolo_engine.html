YoloEngine — YOLO inference engine for smart surveillance

Main features:

  • Lazy loading of model (loads only when needed)
  • scan_single_image: analyze NVR images with high confidence (>=0.60)
  • scan_video: analyze videos with adaptive stride and early exit
  • Day/night thresholds per camera (thresholds / thresholds_night)
  • Stride based on actual video FPS (vid_stride_sec in settings)
  • Initial warmup at reduced stride (first warmup_sec seconds)
  • Dynamic stride for quiet cameras (e.g. parking lot)
  • low_conf_image_scan: low confidence scan for fallback

 80    def _get_thresholds(self, cam_cfg: dict, timestamp: datetime) -> dict:
 81        """Get correct thresholds based on current time (day vs night).
 82        
 83        If camera doesn't have 'thresholds_night', uses day thresholds instead.
 84        Determines day/night based on astronomical sunset/sunrise.
 85        
 86        Args:
 87            cam_cfg: Camera configuration dict
 88            timestamp: Frame timestamp with timezone info
 89            
 90        Returns:
 91            Threshold dictionary (day or night version)
 92        """
 93        thresholds_day   = cam_cfg.get("thresholds", {})
 94        thresholds_night = cam_cfg.get("thresholds_night", thresholds_day)
 95        is_night = is_night_astronomic(timestamp, self.lat, self.lon)
 96        return thresholds_night if is_night else thresholds_day

102    def scan_single_image(self, image_path: Path, video_path: Path,
103                          frames_dir: Path, cam_id: str) -> Optional[dict]:
104        """Analyze a single NVR image with high confidence threshold (>=0.60).
105        
106        Used for fast-track: if NVR image is already sufficient,
107        the video is classified without opening it.
108        
109        Args:
110            image_path: Path to NVR image
111            video_path: Associated video file path
112            frames_dir: Directory for saving extracted frames
113            cam_id: Camera identifier
114            
115        Returns:
116            Detection result dict or None if no promising detections
117        """
118        MIN_CONFIDENCE = 0.60
119        MIN_AREA_RATIO = 0.002
120
121        self.ensure_model_loaded()
122
123        res = self.model(str(image_path), classes=[0], verbose=False, device=self.device)[0]
124        h, w = res.orig_shape
125        img_area = h * w
126
127        best_det = None
128        max_conf  = 0.0
129
130        for box in res.boxes:
131            conf = float(box.conf[0])
132            if conf < MIN_CONFIDENCE:
133                continue
134            x1, y1, x2, y2 = [int(x) for x in box.xyxy[0]]
135            if ((x2 - x1) * (y2 - y1)) / img_area < MIN_AREA_RATIO:
136                continue
137            if conf > max_conf:
138                max_conf   = conf
139                cls_id     = int(box.cls[0])
140                label_name = res.names.get(cls_id, "unknown")
141                best_det   = {
142                    "confidence": conf,
143                    "bbox":       [x1, y1, x2, y2],
144                    "label":      label_name,
145                }
146                
147
148        if not best_det:
149            return None
150
151        # Save crop
152        frames_dir.mkdir(parents=True, exist_ok=True)
153        crop_path = frames_dir / f"{image_path.stem}_crop{image_path.suffix}"
154        try:
155            img = cv2.imread(str(image_path))
156            if img is not None:
157                cx1, cy1, cx2, cy2 = get_crop_coordinates(best_det["bbox"], img.shape)
158                cv2.imwrite(str(crop_path), img[cy1:cy2, cx1:cx2])
159                best_det["crop_path"] = str(crop_path)
160            else:
161                best_det["crop_path"] = None
162        except Exception as e:
163            log.critical(f"Error cropping NVR image: {e}")
164            best_det["crop_path"] = None
165
166        ts = datetime.fromtimestamp(image_path.stat().st_mtime)
167
168        return {
169            "camera_id":        cam_id,
170            "video_path":       str(video_path),
171            "categories_found": ["person"],
172            "frames": [{
173                "category":   "person",
174                "frame_path": str(image_path),
175                "crop_path":  best_det.get("crop_path"),
176                "confidence": best_det["confidence"],
177                "bbox":       best_det["bbox"],
178                "timestamp":  ts.isoformat(),
179            }],
180            "resolved_by": "nvr_image",
181        }

187    def scan_video(self, video_path: Path, frames_dir: Path, cam_id: str) -> Optional[dict]:
188        """
189        Analyze video with adaptive stride and early exit.
190
191        Changes from previous version:
192          - Day/night thresholds per camera (thresholds / thresholds_night)
193          - Stride calculated in real seconds (vid_stride_sec)
194          - Initial warmup with reduced stride (first warmup_sec seconds)
195          - Dynamic stride unchanged (optional per camera)
196        """
197        # ------------------------------------------------------------------
198        # 1. Configurazione e Target IDs
199        # ------------------------------------------------------------------
200        cam_cfg       = self.cameras_config.get(str(cam_id), {})
201        ignore_labels = cam_cfg.get("filters", {}).get("ignore_labels", [])
202
203        self.ensure_model_loaded()
204
205        final_target_ids = get_target_ids(
206            model=self.model,
207            settings=self.settings,
208            mode=self.mode,
209            camera_ignore_labels=ignore_labels,
210        )
211        if not final_target_ids:
212            log.warning(f"No target class for camera={cam_id}")
213            return None
214
215        # ------------------------------------------------------------------
216        # 2. Apertura Video e Parametri Temporali
217        # ------------------------------------------------------------------
218        cap = get_video_capture(video_path)
219        if cap is None:
220            return None
221
222        fps = cap.get(cv2.CAP_PROP_FPS) or 25.0
223
224        # Day/night thresholds based on video mtime
225        video_ts   = datetime.fromtimestamp(video_path.stat().st_mtime)
226        thresholds = self._get_thresholds(cam_cfg, video_ts)
227
228        # Stride in real seconds → frame
229        stride_sec = self.yolo_cfg.get("vid_stride_sec", None)
230        if stride_sec is not None:
231            stride_std = max(1, round(fps * stride_sec))
232        else:
233            stride_std = self.yolo_cfg.get("vid_stride", 12)
234
235        time_gap_sec            = self.yolo_cfg.get("time_gap_sec", 3)
236        occ_gap_frames          = int(fps * time_gap_sec)
237        num_occurrence          = self.yolo_cfg.get("num_occurrence", 3)
238        yolo_override_threshold = self.yolo_cfg.get("yolo_override_threshold", 0.85)
239
240        # ------------------------------------------------------------------
241        # 3. Strutture Dati
242        # ------------------------------------------------------------------
243        detection_groups = self.yolo_cfg.get("detection_groups", {})
244
245        label_to_group = {
246            lbl: group_name.lower()
247            for group_name, labels in detection_groups.items()
248            for lbl in labels
249        }
250
251        detections       = {g.lower(): [] for g in detection_groups}
252        occ_count        = {g.lower(): 0  for g in detection_groups}
253        last_saved_frame = {g.lower(): -occ_gap_frames for g in detection_groups}
254
255        frame_idx             = 0
256        self.last_detection_idx = -9999
257        prev_stride           = stride_std
258
259        # ------------------------------------------------------------------
260        # 4. Analysis Loop
261        # ------------------------------------------------------------------
262        while True:
263            ret, frame = cap.read()
264            if not ret:
265                break
266
267            found_valid_target = False
268
269            yolo_results = self.model.predict(
270                source=frame,
271                classes=final_target_ids,
272                verbose=False,
273                conf=0.20,
274                device=self.device,
275            )[0]
276
277            if len(yolo_results.boxes) > 0:
278                for box in yolo_results.boxes:
279                    cls_id = int(box.cls[0])
280                    conf   = float(box.conf[0])
281                    label  = self.model.names[cls_id]
282                    group  = label_to_group.get(label)
283
284                    if not group or conf < thresholds.get(group, 0.50):
285                        continue
286
287                    found_valid_target      = True
288                    self.last_detection_idx = frame_idx
289
290                    x1, y1, x2, y2 = map(int, box.xyxy[0])
291                    bbox_area  = (x2 - x1) * (y2 - y1)
292                    h, w_frame = frame.shape[:2]
293                    area_ratio = bbox_area / (h * w_frame)
294
295                    detections[group].append({
296                        "frame_idx":     frame_idx,
297                        "confidence":    conf,
298                        "bbox":          [x1, y1, x2, y2],
299                        "area":          bbox_area,
300                        "area_ratio":    area_ratio,
301                        "yolo_label":    label,
302                        "yolo_reliable": conf >= yolo_override_threshold,
303                    })
304
305                    if group in ("person", "animal"):
306                        if frame_idx - last_saved_frame[group] >= occ_gap_frames:
307                            last_saved_frame[group] = frame_idx
308                            occ_count[group] += 1
309
310            # Early exit
311            if occ_count["person"] >= num_occurrence or occ_count["animal"] >= num_occurrence:
312                log.debug(f"Early stop su video={video_path.name}")
313                break
314
315            # Calculate stride for next jump
316            current_stride = self._get_next_stride(
317                frame_idx, fps, found_valid_target, cam_cfg, stride_std
318            )
319            if current_stride != prev_stride:
320                prev_stride = current_stride
321            
322            if current_stride > 1:
323                for _ in range(current_stride - 1):
324                    if not cap.grab():
325                        break
326                    frame_idx += 1
327
328            frame_idx += 1
329
330        cap.release()
331
332        # ------------------------------------------------------------------
333        # 5. Final frame extraction
334        # ------------------------------------------------------------------
335        if not any(detections.values()):
336            return {
337                "camera_id":        cam_id,
338                "video_path":       str(video_path),
339                "categories_found": ["others"],
340                "frames":           [],
341                "timestamp":        datetime.now().isoformat(),
342            }
343
344        frames_dir.mkdir(parents=True, exist_ok=True)
345        cap = cv2.VideoCapture(str(video_path))
346        saved_frames = extract_frames_with_cache(
347            cap=cap,
348            detections=detections,
349            fps=fps,
350            video_path=video_path,
351            frames_dir=frames_dir,
352            frames_per_category=num_occurrence,
353        )
354        cap.release()
355
356        if not saved_frames:
357            return {
358                "camera_id":        cam_id,
359                "video_path":       str(video_path),
360                "categories_found": ["others"],
361                "frames":           [],
362                "timestamp":        datetime.now().isoformat(),
363            }
364
365        categories_found = [cat for cat, found_list in detections.items() if found_list]
366
367        return {
368            "camera_id":        cam_id,
369            "video_path":       str(video_path),
370            "categories_found": categories_found,
371            "frames":           saved_frames,
372            "timestamp":        datetime.now().isoformat(),
373        }

376    # Adaptive stride
377    # ------------------------------------------------------------------
378
379    def _get_next_stride(self, frame_idx: int, fps: float,
380                         found_valid_target: bool, cam_config: dict,
381                         stride_std: int) -> int:
382        """
383        Calculate stride for next frame jump.
384
385        Priority logic:
386          1. Warmup: first warmup_sec seconds → reduced stride (high sensitivity)
387          2. Camera without dynamic_stride → standard stride
388          3. Detection active → standard stride
389          4. Pre-roll (protected zone after warmup) → standard stride
390          5. Cooldown after last detection → standard stride
391          6. Cruising speed → stride_fast
392        """
393        dyn_settings = self.yolo_cfg.get("dynamic_stride_settings", {})
394
395        # 1. Initial warmup: always active, regardless of dynamic_stride
396        warmup_sec = dyn_settings.get("warmup_sec", 5)
397        if frame_idx < int(warmup_sec * fps):
398            return max(1, round(fps * 0.25))  # ~4 frame/sec in first N seconds
399
400        # 2. Without dynamic_stride: standard stride for entire duration
401        if not cam_config.get("dynamic_stride", False):
402            return stride_std
403
404        # 3. Parametri dynamic stride
405        stride_fast_sec = dyn_settings.get("stride_fast_sec", 1.0)
406        stride_fast = max(1, round(fps * stride_fast_sec))
407        pre_roll_sec = dyn_settings.get("pre_roll_sec", 20)
408        cooldown_sec = dyn_settings.get("cooldown_sec", 5)
409
410        # 4. Detection attiva → attenzione massima
411        if found_valid_target:
412            return stride_std
413
414        # 5. Pre-roll (zona protetta dopo il warmup)
415        if frame_idx < int(pre_roll_sec * fps):
416            return stride_std
417
418        # 6. Cooldown after last detection
419        if (frame_idx - self.last_detection_idx) < int(cooldown_sec * fps):
420            return stride_std
421
422        # 7. Cruising speed: nothing found for a while
423        return stride_fast

429    def scan_fallback(self, image_path: Path, target_ids: list) -> Optional[dict]:
430        """
431        High sensitivity scan on NVR image for fallback recovery.
432        Used to recover events that YOLO missed in the first pass.
433        """
434        res = self.model(
435            str(image_path), classes=target_ids,
436            conf=0.15, verbose=False, device=self.device
437        )[0]
438
439        if len(res.boxes) == 0:
440            log.debug(f"YOLO Fallback: nothing found su img={image_path.name}")
441            return None
442
443        best_box = res.boxes[0]
444        return {
445            "confidence":    float(best_box.conf[0]),
446            "bbox":          [int(x) for x in best_box.xyxy[0]],
447            "yolo_reliable": False,
448        }

454    def low_conf_image_scan(self, image_path: Path, video_path: Path,
455                            cam_id: str) -> Optional[dict]:
456        """
457        Very low confidence scan (0.05) on NVR image.
458        Used in advanced fallback phase to recover borderline events.
459        """
460        cam_cfg       = self.cameras_config.get(str(cam_id), {})
461        ignore_labels = cam_cfg.get("filters", {}).get("ignore_labels", [])
462
463        target_ids = get_target_ids(
464            model=self.model,
465            settings=self.settings,
466            mode=self.mode,
467            camera_ignore_labels=ignore_labels,
468        )
469
470        results = self.model(
471            str(image_path), classes=target_ids,
472            conf=0.05, verbose=False, device=self.device
473        )
474
475        if not results or len(results[0].boxes) == 0:
476            log.debug(f"YOLO low_conf: nothing found su img={image_path.name}")
477            return None
478
479        best_det   = None
480        max_conf   = -1.0
481        label_name = "unknown"
482
483        for box in results[0].boxes:
484            conf = float(box.conf[0])
485            if conf > max_conf:
486                max_conf   = conf
487                cls_id     = int(box.cls[0])
488                label_name = self.model.names.get(cls_id, "unknown")
489                best_det   = {
490                    "category":   label_name.lower(),
491                    "confidence": conf,
492                    "bbox":       [int(x) for x in box.xyxy[0]],
493                }
494
495        if not best_det:
496            return None
497
498
499        return {
500            "camera_id":     cam_id,
501            "video_name":    Path(video_path).name,
502            "video_path":    video_path,
503            "image_path":    image_path,
504            "yolo_category": best_det["category"],
505            "yolo_data":     best_det,
506        }

376    # ------------------------------------------------------------------
377    # Adaptive stride
378    # ------------------------------------------------------------------
379
380    def _get_next_stride(self, frame_idx: int, fps: float,
381                         found_valid_target: bool, cam_config: dict,
382                         stride_std: int) -> int:
383        """
384        Calculate stride for next frame jump.
385
386        Priority logic:
387          1. Warmup: first warmup_sec seconds → reduced stride (high sensitivity)
388          2. Camera without dynamic_stride → standard stride
389          3. Detection active → standard stride
390          4. Pre-roll (protected zone after warmup) → standard stride
391          5. Cooldown after last detection → standard stride
392          6. Cruising speed → stride_fast
393        """
394        dyn_settings = self.yolo_cfg.get("dynamic_stride_settings", {})
395
396        # 1. Initial warmup: always active, regardless of dynamic_stride
397        warmup_sec = dyn_settings.get("warmup_sec", 5)
398        if frame_idx < int(warmup_sec * fps):
399            return max(1, round(fps * 0.25))  # ~4 frame/sec in first N seconds
400
401        # 2. Without dynamic_stride: standard stride for entire duration
402        if not cam_config.get("dynamic_stride", False):
403            return stride_std
404
405        # 3. Parametri dynamic stride
406        stride_fast_sec = dyn_settings.get("stride_fast_sec", 1.0)
407        stride_fast = max(1, round(fps * stride_fast_sec))
408        pre_roll_sec = dyn_settings.get("pre_roll_sec", 20)
409        cooldown_sec = dyn_settings.get("cooldown_sec", 5)
410
411        # 4. Detection attiva → attenzione massima
412        if found_valid_target:
413            return stride_std
414
415        # 5. Pre-roll (zona protetta dopo il warmup)
416        if frame_idx < int(pre_roll_sec * fps):
417            return stride_std
418
419        # 6. Cooldown after last detection
420        if (frame_idx - self.last_detection_idx) < int(cooldown_sec * fps):
421            return stride_std
422
423        # 7. Cruising speed: nothing found for a while
424        return stride_fast
