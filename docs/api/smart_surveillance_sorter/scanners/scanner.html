 488    def _get_arbitration_queue(self):
 489        """
 490        Analizza i risultati di YOLO e BLIP per isolare i casi dubbi.
 491        Non servono parametri: legge self.results e self.clip_blip_results.
 492        """
 493        vision_queue = []
 494
 495        # 1. Set of "active" videos in current cycle (full_index + skipped_index)
 496        active_videos = set()
 497        for cam_id, entries in self.full_index.items():
 498            for entry in entries:
 499                active_videos.add(str(entry["video_path"]))
 500        for cam_id, entries in self.skipped_index.items():
 501            for entry in entries:
 502                active_videos.add(str(entry["video_path"]))
 503
 504        # 2. Cache Vision per resume
 505        vision_cache_file = Path(self.output_dir) / VISION_CACHE
 506        processed_vision = {}
 507        if vision_cache_file.exists():
 508            cache_data = load_json(vision_cache_file)
 509            processed_vision = {str(r['video_path']): r for r in cache_data}
 510
 511        # 3. Mappa risultati BLIP
 512        blip_map = {str(r['video_path']): r['category'].lower() for r in self.clip_blip_results}
 513
 514        for video_data in self.results:
 515            video_path = str(video_data.get("video_path"))
 516
 517            # Skip inactive videos (already sorted in previous cycles)
 518            if video_path not in active_videos:
 519                continue
 520
 521            # Vision cache: already processed, update clip_blip_results and skip
 522            if video_path in processed_vision:
 523                cache_entry = processed_vision[video_path]
 524                for i, r in enumerate(self.clip_blip_results):
 525                    if str(r["video_path"]) == video_path:
 526                        self.clip_blip_results[i] = cache_entry
 527                        break
 528                continue
 529
 530            yolo_cats = video_data.get("categories_found", [])
 531            blip_cat  = blip_map.get(video_path, "others")
 532
 533            # YOLO ha visto PERSON → legge, skip
 534            if "person" in yolo_cats:
 535                continue
 536
 537            has_animal  = "animal"  in yolo_cats
 538            has_vehicle = "vehicle" in yolo_cats
 539
 540            if has_animal or has_vehicle:
 541                # CASO A: YOLO vede qualcosa, BLIP dice others
 542                if blip_cat == "others":
 543                    vision_queue.append(video_data)
 544                # CASO B: Conflitto YOLO=animal, BLIP=person
 545                elif has_animal and blip_cat == "person":
 546                    vision_queue.append(video_data)
 547                # CASO C: BLIP conferma animal (recall 9%, vogliamo conferma Qwen)
 548                elif blip_cat == "animal":
 549                    vision_queue.append(video_data)
 550
 551        return vision_queue

 489    def _get_arbitration_queue(self):
 490        """
 491        Analizza i risultati di YOLO e BLIP per isolare i casi dubbi.
 492        Non servono parametri: legge self.results e self.clip_blip_results.
 493        """
 494        vision_queue = []
 495
 496        # 1. Set of "active" videos in current cycle (full_index + skipped_index)
 497        active_videos = set()
 498        for cam_id, entries in self.full_index.items():
 499            for entry in entries:
 500                active_videos.add(str(entry["video_path"]))
 501        for cam_id, entries in self.skipped_index.items():
 502            for entry in entries:
 503                active_videos.add(str(entry["video_path"]))
 504
 505        # 2. Cache Vision per resume
 506        vision_cache_file = Path(self.output_dir) / VISION_CACHE
 507        processed_vision = {}
 508        if vision_cache_file.exists():
 509            cache_data = load_json(vision_cache_file)
 510            processed_vision = {str(r['video_path']): r for r in cache_data}
 511
 512        # 3. Mappa risultati BLIP
 513        blip_map = {str(r['video_path']): r['category'].lower() for r in self.clip_blip_results}
 514
 515        for video_data in self.results:
 516            video_path = str(video_data.get("video_path"))
 517
 518            # Skip inactive videos (already sorted in previous cycles)
 519            if video_path not in active_videos:
 520                continue
 521
 522            # Vision cache: already processed, update clip_blip_results and skip
 523            if video_path in processed_vision:
 524                cache_entry = processed_vision[video_path]
 525                for i, r in enumerate(self.clip_blip_results):
 526                    if str(r["video_path"]) == video_path:
 527                        self.clip_blip_results[i] = cache_entry
 528                        break
 529                continue
 530
 531            yolo_cats = video_data.get("categories_found", [])
 532            blip_cat  = blip_map.get(video_path, "others")
 533
 534            # YOLO ha visto PERSON → legge, skip
 535            if "person" in yolo_cats:
 536                continue
 537
 538            has_animal  = "animal"  in yolo_cats
 539            has_vehicle = "vehicle" in yolo_cats
 540
 541            if has_animal or has_vehicle:
 542                # CASO A: YOLO vede qualcosa, BLIP dice others
 543                if blip_cat == "others":
 544                    vision_queue.append(video_data)
 545                # CASO B: Conflitto YOLO=animal, BLIP=person
 546                elif has_animal and blip_cat == "person":
 547                    vision_queue.append(video_data)
 548                # CASO C: BLIP conferma animal (recall 9%, vogliamo conferma Qwen)
 549                elif blip_cat == "animal":
 550                    vision_queue.append(video_data)
 551
 552        return vision_queue
