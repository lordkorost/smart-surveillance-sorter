 25def load_json(full_path):
 26    """Load JSON data from file.
 27    
 28    Args:
 29        full_path: Path to JSON file (Path object or string)
 30        
 31    Returns:
 32        Loaded JSON data or None if file not found or invalid JSON
 33    """
 34    if not isinstance(full_path, Path):
 35        full_path = PROJECT_ROOT / full_path
 36    try:
 37        with open(full_path, 'r', encoding='utf-8') as f:
 38            return json.load(f)
 39    except FileNotFoundError:
 40        log.critical(f"Not found file={full_path}")
 41        return None
 42    except json.JSONDecodeError:
 43        log.critical(f"File={full_path} is not a valid JSON.")

109def get_crop_coordinates(bbox, frame_shape, margin_perc=1.0):
110    """Calculate crop coordinates with margin from a bounding box.
111    
112    Args:
113        bbox: Bounding box as [x1, y1, x2, y2]
114        frame_shape: Shape of the frame (height, width, ...)
115        margin_perc: Margin percentage relative to bbox size (default: 1.0 = 100%)
116        
117    Returns:
118        List of cropped coordinates [x1, y1, x2, y2] clamped to frame boundaries
119    """
120    x1, y1, x2, y2 = bbox
121    h, w = frame_shape[:2]
122    
123    margin_w = int((x2 - x1) * margin_perc)
124    margin_h = int((y2 - y1) * margin_perc)
125    
126    return [
127        max(x1 - margin_w, 0),
128        max(y1 - margin_h, 0),
129        min(x2 + margin_w, w),
130        min(y2 + margin_h, h)
131    ]

133def get_target_ids(model, settings, mode, camera_ignore_labels):
134    """
135    Retrieve a list of numeric IDs to monitor based on the selected mode,
136    camera‑specific ignore labels, and the set of labels supported by the
137    YOLO model.
138
139    Parameters
140    ----------
141    model : object
142        The YOLO model instance used for detection. Its supported
143        class labels are extracted to determine which IDs are valid
144        for monitoring.
145    settings : dict
146        Configuration dictionary containing YOLO detection settings,
147        including detection groups and other relevant parameters.
148    mode : str
149        Monitoring mode that dictates which categories of labels to
150        consider. Common modes include 'full', 'person', and
151        'person_animal'.
152    camera_ignore_labels : list[str]
153        Labels that should be excluded for a particular camera. These
154        are removed from the candidate label list before the final
155        target ID list is produced.
156
157    Returns
158    -------
159    list[int]
160        A list of numeric class IDs that should be tracked. The IDs
161        are derived from the YOLO model’s supported labels, filtered
162        by the selected mode and any camera‑specific ignore labels.
163
164    Notes
165    -----
166    The method first builds a candidate label list from the active
167    detection groups defined in the settings. It then removes any
168    labels that the camera has marked to ignore, resulting in the
169    final set of labels to monitor. These labels are subsequently
170    mapped to numeric IDs using the YOLO model’s class mapping
171    [1].
172    """
173    # 1. Model name -> ID mapping (e.g. {'person': 0, 'bicycle': 1...})
174    model_name_to_id = {v: k for k, v in model.names.items()}
175    
176    # 2. Define which groups to activate based on MODE
177    mode_map = {
178        "full": ["PERSON", "ANIMAL", "VEHICLE"],
179        "person": ["PERSON"],
180        "person_animal": ["PERSON", "ANIMAL"]
181    }
182    active_groups = mode_map.get(mode, ["PERSON"])
183    
184    # 3. Build list of "candidate" labels from active groups
185    candidate_labels = []
186    #groups_config = settings["yolo_settings"]["detection_groups"]
187    # 1. Retrieve yolo_settings (if missing, use empty dict {})
188    yolo_conf = settings.get("yolo_settings", {})
189
190    # 2. Retrieve detection_groups (if missing, use empty list/dict)
191    groups_config = yolo_conf.get("detection_groups", [])
192
193    for group in active_groups:
194        candidate_labels.extend(groups_config.get(group, []))
195    
196    # 4. Remove camera-specific ignore labels
197    final_labels = [
198        label for label in candidate_labels 
199        if label not in camera_ignore_labels
200    ]
201    
202    # 5. Transform names to numeric IDs (only if they exist in the model)
203    target_ids = [
204        model_name_to_id[name] 
205        for name in final_labels 
206        if name in model_name_to_id
207    ]
208    
209    return list(set(target_ids)) # Return unique IDs

211def calculate_score(category, conf, vision_answer, scoring_settings):
212    """Calculate detection score based on confidence and vision model agreement.
213    
214    Combines YOLO confidence with vision model verification to produce a final score.
215    Applies multipliers and weights based on configuration.
216    
217    Args:
218        category: Detected object category
219        conf: YOLO confidence score (0-1)
220        vision_answer: Vision model's assessment of the detection
221        scoring_settings: Dict with "weights" and "multipliers" configuration
222        
223    Returns:
224        Final score after applying weights and multipliers
225    """
226    weights = scoring_settings.get("weights", {})
227    multipliers = scoring_settings.get("multipliers", {})
228
229    # 1. Base score calculation based on YOLO confidence
230    if conf >= 0.70:
231        base = weights.get("score_high", 3.0)
232    elif conf >= 0.55:
233        base = weights.get("score_mid", 2.0)
234    elif conf >= 0.40:
235        base = weights.get("score_low", 1.0)
236    else:
237        base = 0.3
238
239    # 2. Modifier based on Vision model response
240    if vision_answer == category:
241        # CONFIRMATION: Heavy bonus
242        multiplier = multipliers.get(category, 1.6)
243        base *= multiplier
244    
245    elif vision_answer == "nothing":
246        base = -10.0 
247            
248    else:
249        base *= 0.1
250            
251    return base

377
378    """Check if a directory exists and verify read/write permissions.

380    Args:

401def validate_ollama_setup(vision_settings):

403    
404    Args:

425            log.warning(f"Model={model_required} is not reachable. Did you forget to download/activate it?")
426            
427        log.info(f"✅ Connect to Ollama ({ip}:{port}). Model={model_required}")
428        return True
429
430    except requests.exceptions.ConnectionError:
431        log.info(f"Ollama is not reachable on {ip}:{port}.")
432        return False

434def fetch_coords_logic(city_name):
435    """Fetch geocoordinates for a city using online and offline methods.
436    
437    Attempts to fetch coordinates online first (Geopy), falls back to offline
438    Astral database if no internet, and defaults to Rome if city not found.
439    
440    Args:
441        city_name: Name of the city to find coordinates for
442        
443    Returns:
444        Dictionary with "lat" and "lon" keys
445    """
446    # LEVEL 1: Try online (Geopy)
447    try:
448        # Controllo rapido connessione
449        from geopy.geocoders import Nominatim
450        geolocator = Nominatim(user_agent="security_app", timeout=3)
451        loc = geolocator.geocode(city_name)
452        if loc:
453            return {"lat": loc.latitude, "lon": loc.longitude}
454    except Exception:
455        log.critical(f"No internet access. Searching city='{city_name}' in local db.")
456
457    # LEVEL 2: OFFLINE fallback (Astral Database)
458    try:
459        city_data = lookup(city_name, database())
460        return {"lat": city_data.latitude, "lon": city_data.longitude}
461    except KeyError:
462        log.critical(f"City='{city_name}' not found offline. Fallback on Rome")
463        return {"lat": 41.89, "lon": 12.49} # Default universal fallback

465def get_smart_coordinates(city_from_settings):
466    """Get and cache geocoordinates for the configured city.
467    
468    Checks if coordinates are cached for the current city, returns cached values if valid,
469    otherwise fetches new coordinates and updates cache.
470    
471    Args:
472        city_from_settings: City name from configuration
473        
474    Returns:
475        Tuple of (latitude, longitude)
476    """
477    update_needed = False
478
479    # 1. Load cache (empty dict if file missing or corrupted)
480    cache = load_json(COORDS_CACHE_JSON) or {}
481
482    # 2. Check if city is the same and coordinates exist
483    if cache.get("city_name") == city_from_settings:
484        lat, lon = cache.get("lat"), cache.get("lon")
485        if lat is not None and lon is not None:
486            return lat, lon
487    
488    # 3. If we reach here (different city or missing data), we need to update
489    update_needed = True
490
491    # 2. If needed, fetch new coordinates
492    if update_needed:
493        coords = fetch_coords_logic(city_from_settings)
494        
495        # Prepare data to save
496        new_cache_data = {
497            "city_name": city_from_settings,
498            "lat": coords["lat"],
499            "lon": coords["lon"]
500        }
501        
502        save_json(new_cache_data, COORDS_CACHE_JSON)
503            
504        return coords["lat"], coords["lon"]

506def is_night_astronomic(dt_frame, lat, lon):
507    """Check if a given time is within astronomical night for a location.
508    
509    Determines if the frame timestamp is in astronomical night, applying a
510    20-minute offset before sunset for civil twilight detection.
511    
512    Args:
513        dt_frame: Datetime of the frame (with timezone info)
514        lat: Latitude of the location
515        lon: Longitude of the location
516        
517    Returns:
518        True if the frame is during night, False otherwise
519    """
520    # Observer uses only local mathematics
521    obs = Observer(latitude=lat, longitude=lon)
522    
523    # Calculate sun position for the video date
524    s = sun(obs, date=dt_frame.date())
525    
526    # Apply 20 min offset (civil twilight)
527    # Detection boost activates 20 min before astronomical sunset
528    sunset_limit = s["sunset"] - timedelta(minutes=20)
529    sunrise_limit = s["sunrise"] + timedelta(minutes=20)
530    
531    # Convert frame timestamp to UTC for comparison
532    dt_utc = dt_frame.astimezone(pytz.utc)
533    
534    return dt_utc > sunset_limit or dt_utc < sunrise_limit
